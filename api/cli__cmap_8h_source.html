156 static inline unsigned int
157 cmap_cnt(lcore_t *lc, getter_fn get)
158 {
159  unsigned cnt = 0;
160 
161  if (!get)
162  return cnt;
163 
164  while (lc) {
165  if (cnt < get(lc))
166  cnt = get(lc);
167  lc = lc->next;
168  }
169  return cnt + 1;
170 }
