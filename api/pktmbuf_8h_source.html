474 static inline uint16_t
475 pktmbuf_refcnt_update(pktmbuf_t *m, int16_t value)
476 {
477  uint16_t refcnt;
478 
479  /*
480  * The atomic_add is an expensive operation, so we don't want to
481  * call it in the case where we know we are the unique holder of
482  * this mbuf (i.e. ref_cnt == 1). Otherwise, an atomic
483  * operation has to be used because concurrent accesses on the
484  * reference counter can occur.
485  */
486  refcnt = pktmbuf_refcnt_read(m);
487  if (likely(refcnt == 1)) {
488  refcnt = refcnt + value;
489  pktmbuf_refcnt_set(m, refcnt);
490  return (uint16_t)refcnt;
491  }
492 
493  __pktmbuf_refcnt_update(m, value);
494 
495  return pktmbuf_refcnt_read(m);
496 }

735 static __cne_always_inline void
736 __pktmbuf_free_bulk(pktmbuf_pending_t *p, pktmbuf_t *m)
737 {
738  m = pktmbuf_refcnt_free(m);
739 
740  if (likely(m != NULL)) {
741  if (p->nb_pending == p->pending_sz || (p->nb_pending > 0 && m->pooldata != p->pooldata)) {
742  __pktmbuf_flush_pending(p);
743  p->pooldata = m->pooldata;
744  }
745 
746  p->pending[p->nb_pending++] = m;
747  }

759 static inline void
760 pktmbuf_free_bulk(pktmbuf_t **mbufs, unsigned int count)
761 {
762  pktmbuf_pending_t pend;
763  memset(&pend, 0, sizeof(pend));
764  pend.pending_sz = PKTMBUF_PENDING_SZ;
765 
766  if (count == 0)
767  return;
768 
769  pend.pooldata = mbufs[0]->pooldata;
770 
771  for (unsigned int idx = 0; idx < count; idx++)
772  __pktmbuf_free_bulk(&pend, mbufs[idx]);
773 
774  if (pend.nb_pending)
775  __pktmbuf_flush_pending(&pend);
776 }
