790 static inline uint64_t
791 cne_str_to_size(const char *str)
792 {
793  char *endptr;
794  unsigned long long size;
795 
796  while (isspace((int)*str))
797  str++;
798  if (*str == '-')
799  return 0;
800 
801  errno = 0;
802  size = strtoull(str, &endptr, 0);
803  if (errno)
804  return 0;
805 
806  if (*endptr == ' ')
807  endptr++; /* allow 1 space gap */
808 
809  switch (*endptr) {
810  case 'G':
811  case 'g':
812  size *= 1024; /* fall-through */
813  case 'M':
814  case 'm':
815  size *= 1024; /* fall-through */
816  case 'K':
817  case 'k':
818  size *= 1024; /* fall-through */
819  default:
820  break;
821  }
822  return size;
823 }
