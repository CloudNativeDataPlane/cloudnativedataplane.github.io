315 static inline int
316 is_match(const char *s1, const char *s2)
317 {
318  if (!s1 || !s2)
319  return 0;
320 
321  while ((*s1 != '\0') && (*s2 != '\0')) {
322  if (*s1++ != *s2++)
323  return 0;
324  }
325  if (*s1 != *s2)
326  return 0;
327 
328  return 1;
329 }

475 {
