106 static inline int
107 cne_net_intel_cksum_flags_prepare(pktmbuf_t *m, uint64_t ol_flags)
108 {
109  /* Initialise ipv4_hdr to avoid false positive compiler warnings. */
110  struct cne_ipv4_hdr *ipv4_hdr = NULL;
111  struct cne_ipv6_hdr *ipv6_hdr;
112  struct cne_tcp_hdr *tcp_hdr;
113  struct cne_udp_hdr *udp_hdr;
114  uint64_t inner_l3_offset = m->l2_len;
115 
116  /*
117  * Does packet set any of available offloads?
118  * Mainly it is required to avoid fragmented headers check if
119  * no offloads are requested.
120  */
121  if (!(ol_flags & (CNE_MBUF_F_TX_IP_CKSUM | CNE_MBUF_F_TX_L4_MASK | CNE_MBUF_F_TX_TCP_SEG)))
122  return 0;
123 
124  if (ol_flags & (CNE_MBUF_F_TX_OUTER_IPV4 | CNE_MBUF_F_TX_OUTER_IPV6))
125  inner_l3_offset += m->outer_l2_len + m->outer_l3_len;
126 
127  /*
128  * Check if headers are fragmented.
129  * The check could be less strict depending on which offloads are
130  * requested and headers to be used, but let's keep it simple.
131  */
132  if (unlikely(pktmbuf_data_len(m) < inner_l3_offset + m->l3_len + m->l4_len))
133  return -ENOTSUP;
134 
135  if (ol_flags & CNE_MBUF_F_TX_IPV4) {
136  ipv4_hdr = pktmbuf_mtod_offset(m, struct cne_ipv4_hdr *, inner_l3_offset);
137 
138  if (ol_flags & CNE_MBUF_F_TX_IP_CKSUM)
139  ipv4_hdr->hdr_checksum = 0;
140  }
141 
142  if ((ol_flags & CNE_MBUF_F_TX_L4_MASK) == CNE_MBUF_F_TX_UDP_CKSUM) {
143  if (ol_flags & CNE_MBUF_F_TX_IPV4) {
144  udp_hdr = (struct cne_udp_hdr *)((char *)ipv4_hdr + m->l3_len);
145  udp_hdr->dgram_cksum = cne_ipv4_phdr_cksum(ipv4_hdr, ol_flags);
146  } else {
147  ipv6_hdr = pktmbuf_mtod_offset(m, struct cne_ipv6_hdr *, inner_l3_offset);
148  /* non-TSO udp */
149  udp_hdr = pktmbuf_mtod_offset(m, struct cne_udp_hdr *, inner_l3_offset + m->l3_len);
150  udp_hdr->dgram_cksum = cne_ipv6_phdr_cksum(ipv6_hdr, ol_flags);
151  }
152  } else if ((ol_flags & CNE_MBUF_F_TX_L4_MASK) == CNE_MBUF_F_TX_TCP_CKSUM ||
153  (ol_flags & CNE_MBUF_F_TX_TCP_SEG)) {
154  if (ol_flags & CNE_MBUF_F_TX_IPV4) {
155  /* non-TSO tcp or TSO */
156  tcp_hdr = (struct cne_tcp_hdr *)((char *)ipv4_hdr + m->l3_len);
157  tcp_hdr->cksum = cne_ipv4_phdr_cksum(ipv4_hdr, ol_flags);
158  } else {
159  ipv6_hdr = pktmbuf_mtod_offset(m, struct cne_ipv6_hdr *, inner_l3_offset);
160  /* non-TSO tcp or TSO */
161  tcp_hdr = pktmbuf_mtod_offset(m, struct cne_tcp_hdr *, inner_l3_offset + m->l3_len);
162  tcp_hdr->cksum = cne_ipv6_phdr_cksum(ipv6_hdr, ol_flags);
163  }
164  }
165 
166  return 0;
167 }
