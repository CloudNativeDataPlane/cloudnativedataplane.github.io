394 static inline void *
395 vec_pool_alloc(mempool_t *mp, int dont_free)
396 {
397  void *v = NULL;
398 
399  if (mp && mempool_get(mp, (void **)&v) == 0) {
400  vec_hdr_t *vec;
401 
402  /* move pointer to the vector data */
403  v = CNE_PTR_ADD(v, CNE_CACHE_LINE_SIZE);
404 
405  /* Move backward to the vector header structure and apply flags */
406  vec = vec_header(v);
407  vec->flags = (dont_free) ? VEC_DONT_FREE_FLAG : 0;
408  } else
409  CNE_ERR("Unable to allocate from vec_pool mp %p\n", mp);
410  return v;
411 }

419 static inline void
420 vec_pool_free(void *vec)
421 {
422  if (vec) {
423  vec_hdr_t *h = vec_header(vec);
424 
425  h->len = 0;
426 
427  if ((h->flags & VEC_DONT_FREE_FLAG) == 0) {
428  if (h->flags & VEC_FREE_FLAG) {
429  CNE_WARN("vec_pool_free() already freed\n");
430  vec_dump(__func__, vec);
431  return;
432  }
433 
434  if (!h->pool)
435  CNE_RET("Calling vec_pool_free() pool pointer being NULL\n");
436 
437  h->flags = VEC_FREE_FLAG;
438  mempool_put(h->pool, CNE_PTR_SUB(vec, CNE_CACHE_LINE_SIZE));
439  }
440  }
