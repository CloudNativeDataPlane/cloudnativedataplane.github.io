273 static __inline__ struct ether_addr *
274 cne_ether_aton(const char *a, void *_e)
275 {
276  int i;
277  char *end;
278  unsigned long o[ETH_ALEN];
279  struct ether_addr *e = _e;
280  static struct ether_addr ether_addr;
281 
282  if (!e)
283  e = &ether_addr;
284 
285  i = 0;
286  do {
287  errno = 0;
288  o[i] = strtoul(a, &end, 16);
289  if (errno != 0 || end == a || (end[0] != ':' && end[0] != 0))
290  return NULL;
291  a = end + 1;
292  } while (++i != sizeof(o) / sizeof(o[0]) && end[0] != 0);
293 
294  /* Junk at the end of line */
295  if (end[0] != 0)
296  return NULL;
297 
298  /* Support the format XX:XX:XX:XX:XX:XX */
299  if (i == ETH_ALEN) {
300  while (i-- != 0) {
301  if (o[i] > UINT8_MAX)
302  return NULL;
303  e->ether_addr_octet[i] = (uint8_t)o[i];
304  }
305  /* Support the format XXXX:XXXX:XXXX */
306  } else if (i == ETH_ALEN / 2) {
307  while (i-- != 0) {
308  if (o[i] > UINT16_MAX)
309  return NULL;
310  e->ether_addr_octet[i * 2] = (uint8_t)(o[i] >> 8);
311  e->ether_addr_octet[i * 2 + 1] = (uint8_t)(o[i] & 0xff);
312  }
313  /* unknown format */
314  } else
315  return NULL;
316 
317  return e;
318 }
