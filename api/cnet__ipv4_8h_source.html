154 /*
155  * Adjust the checksum to reflect that the TTL had been decremented.
156  *
157  * Flip the bits on the checksum, decrement the high byte of the checksum,
158  * fold in any carry, and then flip the bits back. Rather than convert
159  * the checksum to host byte order and then back to network byte order,
160  * just convert the increment to network byte order. Note: in 1's
161  * complement arithmetic, subtracting by x is the same as adding the 1's
162  * complement of x. So, in 16 bit arithmetic, rather than subtracting by
163  * (1<<8), we can add by (1<<8)^0xffff. Since it's all constants, that
164  * should be evaluated by the compiler at compile time.
165  *
166  * Doing the ^0xffff to initially flip the bits keeps the upper bits from
167  * also being flipped. Using the ~ operation at the end doesn't matter,
168  * because the upper bits get tossed when we assign it to the 16 bit sum
169  * field, so let the compiler do whatever is fastest.
170  */
171 static inline void
172 ipv4_adjust_cksum(struct cne_ipv4_hdr *hdr)
173 {
174  int32_t cksum;
175 
176  hdr->time_to_live--;
177 
178  /* increment checksum high byte */
179  cksum = (int32_t)(hdr->hdr_checksum ^ 0xFFFF) + (int32_t)htobe16(((1 << 8) ^ 0xFFFF));
180 
181  /* Fold the carry bit into the checksum */
182  hdr->hdr_checksum = ~(cksum + (cksum >> 16));
183 }
