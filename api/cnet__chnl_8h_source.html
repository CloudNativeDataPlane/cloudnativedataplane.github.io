390 static inline int
391 ch_readable(struct chnl *ch)
392 {
393  int ret;
394 
395  if (!ch) {
396  CNE_ERR("Channel is NULL\n");
397  return 0;
398  }
399 
400  ret =
401  (cb_avail(&ch->ch_rcv) >= ch->ch_rcv.cb_lowat) && !chnl_cant_rcv_more(ch) && !ch->ch_error;
402 
403  return ret;
404 }

414 static inline int
415 ch_writeable(struct chnl *ch)
416 {
417  int ret;
418 
419  if (!ch) {
420  CNE_ERR("Channal is NULL\n");
421  return 0;
422  }
423 
424  ret = (cb_space(&ch->ch_snd) >= ch->ch_snd.cb_lowat &&
425  ((ch->ch_state & _ISCONNECTED) || (ch->ch_proto->proto != SOCK_STREAM))) ||
426  !chnl_cant_snd_more(ch) || !ch->ch_error;
427 
428  return ret;
429 }
